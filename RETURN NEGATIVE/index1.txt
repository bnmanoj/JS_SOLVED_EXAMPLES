<script>
  // Function to return the negative of a number
  function returnNegative(number) {
    // If number is positive or zero, return its negative value
    if (number >= 0) {
      return -number;
    } else {
      // If number is negative, return the number itself
      return number;
    }
  }

  // Example usage
  const num1 = 5;
  const num2 = -7;
  console.log("Negative of", num1, "is", returnNegative(num1));
  console.log("Negative of", num2, "is", returnNegative(num2));
</script>
