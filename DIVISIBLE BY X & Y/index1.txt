<script>
  // Function to check if a number is divisible by both X and Y
  function isDivisible(number, x, y) {
    return number % x === 0 && number % y === 0;
  }

  // Example usage
  const number = 12;
  const x = 3;
  const y = 4;
  if (isDivisible(number, x, y)) {
    console.log(`${number} is divisible by both ${x} and ${y}.`);
  } else {
    console.log(`${number} is not divisible by both ${x} and ${y}.`);
  }
</script>
