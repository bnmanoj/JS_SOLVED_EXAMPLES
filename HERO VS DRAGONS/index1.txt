<script>
  // Function to determine if a hero can defeat a dragon
  function heroVsDragon(heroAttack, dragonHealth) {
    if (heroAttack >= dragonHealth) {
      return "Hero wins!";
    } else {
      return "Dragon wins!";
    }
  }

  // Example usage
  const heroAttack = 50; // Hero's attack power
  const dragonHealth = 100; // Dragon's health
  const result = heroVsDragon(heroAttack, dragonHealth);
  console.log(result);
</script>
